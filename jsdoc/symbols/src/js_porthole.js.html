100 /**
101  * Legacy browser implementation of proxy window object to post message to target window
102  *
103  * @private
104  * @constructor
105  * @param proxyIFrameUrl
106  * @param targetWindowName
107  */
108 Porthole.WindowProxyLegacy = function(proxyIFrameUrl, targetWindowName) {
109 	if (targetWindowName == null) {
110 		targetWindowName = '';
111 	}
112 	this.targetWindowName = targetWindowName;
113 
114 	this.eventListeners = [];
115 	this.origin = window.location.protocol + '//' + window.location.host
116 	if (proxyIFrameUrl != null) {	
117 		this.proxyIFrameName = this.targetWindowName + 'ProxyIFrame';
118 		this.proxyIFrameLocation = proxyIFrameUrl;
119 
120 		// Create the proxy iFrame and add to dom
121 		this.proxyIFrameElement = this.createIFrameProxy();
122 	} else {
123 		// Won't be able to send messages
124 		this.proxyIFrameElement = null;
125 	}
126 }
127 
128 Porthole.WindowProxyLegacy.prototype = {
129 	getTargetWindowName: function() {
130 		return this.targetWindowName;
131 	},
132 	
133 	getOrigin: function() {
134 		return this.origin;
135 	},
136 	
137 	/**
138 	 * Create an iframe and load the proxy
139 	 *
140 	 * @private
141 	 * @returns iframe
142 	 */
143 	createIFrameProxy: function() {
144 		var iframe = document.createElement("iframe");
145 		iframe.setAttribute('id', this.proxyIFrameName);
146 		iframe.setAttribute('name', this.proxyIFrameName);
147 		iframe.setAttribute('src', this.proxyIFrameLocation);
148 		iframe.setAttribute('frameBorder', '1'); // IE needs this otherwise resize event is not fired
149 		iframe.setAttribute('scrolling', 'auto');
150 		iframe.setAttribute('width', 30);	// Need a certain size othwerise IE7 does not fire resize event
151 		iframe.setAttribute('height', 30);
152 		iframe.setAttribute('style', "position: absolute; left: -100px; top:0px;");
153 		// IE needs this because setting style attribute is broken. No really.
154 		if (iframe.style.setAttribute) {
155 			iframe.style.setAttribute('cssText', "position: absolute; left: -100px; top:0px;");
156 		}
157 		document.body.appendChild(iframe);
158 		return iframe;
159 	},
160 	
161 	postMessage: function(data, targetOrigin) {
162 		if (targetOrigin == null) {
163 			targetOrigin = '*';
164 		}
165 		if (this.proxyIFrameElement == null) {
166 			error("Can't send message because no proxy url was passed in the constructor");
167 		} else {
168 			sourceWindowName = window.name;
169 			this.proxyIFrameElement.setAttribute('src', this.proxyIFrameLocation + '#' + data +
170 				'&sourceOrigin=' + escape(this.getOrigin()) + 
171 				'&targetOrigin=' + escape(targetOrigin) + 
172 				'&sourceWindowName=' + sourceWindowName + 
173 				'&targetWindowName=' + this.targetWindowName);
174 			this.proxyIFrameElement.height = this.proxyIFrameElement.height > 50 ? 50 : 100;
175 		}
176 	},
177 
178 	addEventListener: function(f) {
179 		this.eventListeners.push(f);
180 		return f;
181 	},
182 
183 	removeEventListener: function(f) {
184 		try {
185 			var index = this.eventListeners.indexOf(f);
186 			this.eventListeners.splice(index, 1);
187 		} catch(e) {
188 			this.eventListeners = [];
189 			error(e);
190 		}
191 	},
192 
193 	dispatchEvent: function(e) {
194 		for (var i = 0; i < this.eventListeners.length; i++) {
195 			try {
196 	    	this.eventListeners[i](e);
197 			} catch(ex) {
198 				error("Exception trying to call back listener: " + ex);
199 			}
200 	  }
201 	}
202 };

204 /**
205 	* Implementation for modern browsers that supports it
206 	*/
207 Porthole.WindowProxyHTML5 = function(proxyIFrameUrl, targetWindowName) {
208 	if (targetWindowName == null) {
209 		targetWindowName = '';
210 	}
211 	this.targetWindowName = targetWindowName;
212 }
213 
214 Porthole.WindowProxyHTML5.prototype = {
215 	postMessage: function(data, targetOrigin) {
216 		if (targetOrigin == null) {
217 			targetOrigin = '*';
218 		}
219 		
220 		// Lookup window object from target window name
221 		if (this.targetWindowName == '') {
222 			targetWindow = top;
223 		} else {
224 			targetWindow = parent.frames[this.targetWindowName];
225 		}
226 		targetWindow.postMessage(data, targetOrigin);
227 	},
228 	addEventListener: function(f) {
229 		window.addEventListener("message", f, false);
230 		return f;
231 	},
232 	removeEventListener: function(f) {
233 		window.removeEventListener("message", f, false);
234 	},
235 	dispatchEvent: function(e) {
236 		var evt = document.createEvent("MessageEvent");
237 		evt.initMessageEvent("message", true, true, e.data, e.origin, 1, window, null);
238 		window.dispatchEvent(evt);
239 	}
240 }

243 	trace("Using legacy browser support");
244 	Porthole.WindowProxy = Porthole.WindowProxyLegacy;

247 	trace("Using legacy browser support");
248 	Porthole.WindowProxy = Porthole.WindowProxyHTML5;

252 /**
253 	* Convinience method to split a message of type param=value¶m2=value2
254 	* and return an array such as ['param':value, 'param2':value2]
255 	*
256 	* @param {String} message
257 	* @returns {Array} key value pair array
258 	*/
259 Porthole.WindowProxy.splitMessageParameters = function(message) {
260 	if (typeof message == 'undefined' || message == null) {
261 		return null;
262 	}
263 	var hash = new Array();
264 	var pairs = message.split(/&/);
265 	for (var keyValuePairIndex in pairs) {
266 		var nameValue = pairs[keyValuePairIndex].split(/=/);
267 		if (typeof(nameValue[1]) == 'undefined') {
268 			hash[nameValue[0]] = '';
269 		} else {
270 			hash[nameValue[0]] = nameValue[1];
271 		}
272 	}
273 	return hash;
274 }

288 /**
289 	* Dispatcher object to relay messages.
290 	* @public
291 	* @constructor
292 	*/
293 Porthole.WindowProxyDispatcher = {
294 	/**
295 		* Forward a message event to the target window
296 		* @private
297 		*/
298 	forwardMessageEvent: function(e) {
299 		//trace("Porthole.WindowProxyDispatcher.forwardMessageEvent");
300 		var message = document.location.hash;
301 		if (message.length > 0) {
302 			// Eat the hash character
303 			message = message.substr(1);
304 		
305 			m = Porthole.WindowProxyDispatcher.parseMessage(message);
306 
307 			if (m.targetWindowName == '') {
308 				targetWindow = top;
309 			} else {
310 				targetWindow = parent.frames[m.targetWindowName];
311 			}
312 		
313 			var windowProxy = Porthole.WindowProxyDispatcher.findWindowProxyObjectInWindow(targetWindow, m.sourceWindowName);
314 
315 			if (windowProxy) {
316 				if (windowProxy.origin == m.targetOrigin || m.targetOrigin == '*') {
317 					e = new Porthole.MessageEvent(m.data, m.sourceOrigin, windowProxy);
318 					windowProxy.dispatchEvent(e);
319 				} else {
320 					error("Target origin " + windowProxy.origin + " does not match desired target of " + m.targetOrigin);
321 				}
322 			} else {
323 				error("Could not find window proxy object on the target window");
324 			}
325 		}
326 	},
327 
328 	parseMessage: function(message) {
329 		if (typeof message == 'undefined' || message == null) {
330 			return null;
331 		}
332 		params = Porthole.WindowProxy.splitMessageParameters(message);
333 		var h = {targetOrigin:'', sourceOrigin:'', sourceWindowName:'', data:''};
334 		h.targetOrigin = unescape(params['targetOrigin']);
335 		h.sourceOrigin = unescape(params['sourceOrigin']);
336 		h.sourceWindowName = unescape(params['sourceWindowName']);
337 		h.targetWindowName = unescape(params['targetWindowName']);
338 		var d = message.split(/&/);
339 		if (d.length > 3) {
340 			d.pop();
341 			d.pop();
342 			d.pop();
343 			d.pop();
344 			h.data = d.join('&');
345 		}
346 		return h;
347 	},
348 	
349 	/**
350 		* Look for a window proxy object in the target window
351 		* @private
352 		*/
353 	findWindowProxyObjectInWindow: function(w, sourceWindowName) {
354 		//trace("Porthole.WindowProxyDispatcher.findWindowProxyObjectInWindow");
355 		// IE does not enumerate global objects on the window object
356 		if (w.RuntimeObject) {
357 			w = w.RuntimeObject();
358 		}
359 		if (w) {
360 			for (var i in w) { 
361 				try {
362 					// Ensure that we're finding the proxy object that is declared to be targetting the window that is calling us
363 					if (w[i] != null && typeof w[i] == "object" && w[i] instanceof w.Porthole.WindowProxy && w[i].getTargetWindowName() == sourceWindowName) { 
364 						return w[i];
365 					}
366 				} catch(e) {
367 					// Swallow exception in case we access an object we shouldn't
368 				}
369 			}
370 		}
371 		return null;
372 	},
373 	
374 	/**
375 	 * Start a proxy to relay messages.
376 	 * @public
377 	 */
378 	start: function() {
379 		if (window.addEventListener) {
380 			window.addEventListener('resize', Porthole.WindowProxyDispatcher.forwardMessageEvent, false); 
381 		} else if (document.body.attachEvent) {
382 			window.attachEvent('onresize', Porthole.WindowProxyDispatcher.forwardMessageEvent);
383 		} else {
384 			// Should never happen
385 			error("Can't attach resize event");
386 		}
387 	}
388 };
